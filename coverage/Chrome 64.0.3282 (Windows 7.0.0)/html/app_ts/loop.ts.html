  private ticker: {addCallback: {(func, scope): void;} }

    constructor(ticker:   {addCallback: {(func, scope): void;} } ){
    this.ticker = ticker;
    this.ticker.addCallback(this._handle, this);
  }
 
  private _handle(){
    for(var component of this._registeredComponents){
          component.draw();

}
